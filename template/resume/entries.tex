% CONFIG %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newlength{\entrysep}
\newlength{\entrylinesep}

\setlength{\entrysep}{1\baselineskip}
\setlength{\entrylinesep}{0.3\baselineskip}

% HEADER %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entryheader}[2]{%
    {\textcolor{ink}{#1}\hfill{\textcolor{muted}{#2}}}% \footnotesize
    \par\vspace{\entrylinesep}%
}

% LINE %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entryline}[1]{%
    #1\par\vspace{\entrylinesep}% \footnotesize
}

% SINGLE LINE %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entrycolumns}[2]{%
    \par\noindent%
    {#1\hspace{0.6em}#2}% \footnotesize
    \par\vspace{\entrylinesep}%
}

% BODY %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entrybody}[1]{%
    {\setlength{\parskip}{0pt}\setlength{\parindent}{0pt}#1}%
}

% ENTRY %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\sectionentry}[3]{%
    \par\noindent
    \entryheader{#1}{#2}%
    \entrybody{#3}%
    \vspace{\entrysep}%
}
