% CONFIG %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newlength{\entrysep}
\newlength{\entrylinesep}

\setlength{\entrysep}{1.6\baselineskip}
\setlength{\entrylinesep}{0.4\baselineskip}

% HEADER %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entryheader}[2]{%
    \par\noindent%
    {\textcolor{ink}{#1}\hfill{\textcolor{muted}{#2}}}%
    \par%
}

% LINE %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entryline}[1]{%
    \par\noindent\addvspace{\entrylinesep}%
    #1%
    \par%
}

% COLUMNS %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entrycolumns}[3][4em]{% left width, left content, right content
    \entryline{\makebox[#1][l]{#2}\hspace{0.6em}#3}%
}

% BODY %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entrybody}[1]{%
    {\setlength{\parskip}{0pt}\setlength{\parindent}{0pt}#1}%
}

% ENTRY %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\makeatletter
\newcommand{\sectionentry}{\@ifstar\sectionentry@star\sectionentry@nostar}

\newcommand{\sectionentry@star}[3]{%
    \par\noindent%
    \entryheader{#1}{#2}%
    \entrybody{#3}%
}

\newcommand{\sectionentry@nostar}[3]{%
    \par\noindent%
    \addvspace{\entrysep}%
    \sectionentry@star{#1}{#2}{#3}%
}
\makeatother
