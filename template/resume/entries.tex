% CONFIG %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newlength{\entrysep}
\setlength{\entrysep}{0.60\baselineskip}

% HEADER %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entryheader}[2]{%
    {\textbf{\textcolor{ink}{#1}}\hfill{\textcolor{muted}{#2}}}\par% \footnotesize
}

% BODY %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entrybody}[1]{%
    {\setlength{\parskip}{0pt}\setlength{\parindent}{0pt}#1}%
}

% ENTRY %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\sectionentry}[3]{%
    \par\noindent
    \entryheader{#1}{#2}%
    \entrybody{#3}%
    \vspace{\entrysep}%
}

% LINE %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entrylinesep}{0.22\baselineskip}
\newcommand{\entryline}[1]{%
    {\textcolor{ink}{#1}}\par\vspace{\entrylinesep}% \footnotesize
}

% SINGLE LINE %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\entrycolumns}[2]{%
    \par\noindent%
    {\textbf{\textcolor{ink}{#1}}\hspace{0.75em}\textcolor{ink}{#2}}\par% \footnotesize
    \vspace{0.30\baselineskip}%
}
