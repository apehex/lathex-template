\usepackage{titlesec}

%----------------------------------------------------------------------------------------
%   SECTION NUMBERING IN THE MARGIN
%----------------------------------------------------------------------------------------

\newcommand{\autodot}{.} % separate title numbers with dots
\newlength{\titlenumbermajoroffset}
\newlength{\titlenumberminoroffset}
\setlength{\titlenumbermajoroffset}{-16mm}
\setlength{\titlenumberminoroffset}{-11mm}
\newcommand{\titlenumberfontsizept}{14}
\makeatletter
\newcommand*{\formatheadnumber}[1]{\textcolor{fg}{\fontsize{\titlenumberfontsizept}{\titlenumberfontsizept}\selectfont\headingfontfamily\bfseries\selectfont#1}}
\newcommand*{\splitheadnumber}[1]{\expandafter\splitheadnumberparts#1.\@nil}
\def\splitheadnumberparts#1.#2\@nil{%
  \makebox[0pt][l]{\hspace*{\titlenumbermajoroffset}\formatheadnumber{#1\autodot}}%
  \if\relax\detokenize{#2}\relax
  \else
    \makebox[0pt][l]{\hspace*{\titlenumberminoroffset}\formatheadnumber{#2\autodot}}%
  \fi
}
\makeatother

%----------------------------------------------------------------------------------------
%   LAYOUT
%----------------------------------------------------------------------------------------

\titlespacing*{\chapter}{0pt}{-30pt}{20pt}

\titlespacing*{\paragraph}{0pt}{3.25ex plus 1ex minus .2ex}{1.5ex plus .2ex}

%----------------------------------------------------------------------------------------
%   FORMAT
%----------------------------------------------------------------------------------------

\titleformat{\chapter}[hang]
  {\normalfont\huge\headingfontfamily\bfseries\selectfont\color{accent}}
  {\splitheadnumber{\thechapter}}
  {0pt}
  {\MakeUppercase}

\titleformat{\section}
  {\normalfont\large\headingfontfamily\bfseries\selectfont\color{fg}}
  {\splitheadnumber{\thesection}}
  {0pt}
  {\MakeUppercase}

\titleformat{\subsection}
  {\normalfont\normalsize\headingfontfamily\bfseries\selectfont\color{fg}}
  {\splitheadnumber{\thesubsection}}
  {0pt}
  {}

\titleformat{\subsubsection}
  {\normalfont\small\headingfontfamily\bfseries\selectfont\color{fg}}
  {\splitheadnumber{\thesubsubsection}}
  {0pt}
  {}

%----------------------------------------------------------------------------------------
%   SPECIAL TITLE
%----------------------------------------------------------------------------------------

\renewcommand{\paragraph}[1]{%
  \par\vspace{3.25ex plus 1ex minus .2ex}%
  \noindent\hspace*{-2cm}\colorbox{bgacc}{%
    \hspace*{2cm}\normalfont\huge\headingfontfamily\selectfont\color{fgacc}\uppercase{#1}
  }%
  \par\nobreak\vspace{1.5ex plus .2ex}%
}
