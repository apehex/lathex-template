\usepackage{eso-pic}

%----------------------------------------------------------------------------------------
%   SHOW / HIDE PART TITLE IN THE LEFT BAR
%----------------------------------------------------------------------------------------

\newif\ifbackgroundtitlevisible
\backgroundtitlevisibletrue

\newif\ifbackgroundbarvisible
\backgroundbarvisibletrue

%----------------------------------------------------------------------------------------
%   CONFIGURE LEFT BAR
%----------------------------------------------------------------------------------------

\newcommand{\leftbarcontents}{%
  \ifbackgroundbarvisible
    \fill[bgacc] (current page.south west) rectangle ([xshift=8mm]current page.north west);
  \fi
  \ifbackgroundtitlevisible
    \node[rotate=90, fgacc, font=\huge\bfseries\headingfontfamily, anchor=west] at ([xshift=4mm,yshift=16mm]current page.south west) {\currentparttitle};
  \fi
}

\AddToShipoutPictureBG{%
  \begin{tikzpicture}[remember picture,overlay]
    \leftbarcontents
  \end{tikzpicture}%
}
